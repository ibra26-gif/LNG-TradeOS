  window.setPublicTheme=function(theme){
    ensurePublicLightStyles();
    const site=document.getElementById('public-site');
    const next=theme==='light'?'light':'dark';
    if(site){
      site.dataset.publicTheme=next;
      site.classList.toggle('public-theme-light',next==='light');
      site.classList.toggle('public-theme-dark',next==='dark');
    }
    document.documentElement.dataset.theme=next;
    document.body.classList.toggle('public-theme-light',next==='light');
    document.body.classList.toggle('public-theme-dark',next==='dark');
    try{localStorage.setItem('lng_public_theme',next);localStorage.setItem('lng_theme',next);}catch{}
    document.getElementById('theme-choice-dark')?.classList.toggle('active',next==='dark');
    document.getElementById('theme-choice-light')?.classList.toggle('active',next==='light');
    const toggle=document.getElementById('public-theme-toggle');
    if(toggle)toggle.textContent='Theme';
    const name=document.getElementById('theme-current-name');
    const note=document.getElementById('theme-current-note');
    if(name)name.textContent=next==='light'?'Steel Light':'Graphite Command';
    if(note)note.textContent=next==='light'?'Steel Light identity active.':'Graphite Command identity active.';
  };
  window.togglePublicTheme=function(){
    const site=document.getElementById('public-site');
    const current=site?.dataset.publicTheme||'dark';
    setPublicTheme(current==='light'?'dark':'light');
  };
  window.publicGo=function(page){
    document.querySelectorAll('.public-page').forEach(el=>el.classList.remove('active'));
    const target=document.getElementById('public-page-'+page)||document.getElementById('public-page-home');
    target.classList.add('active');
    document.querySelectorAll('[data-public-nav]').forEach(btn=>btn.classList.toggle('active',btn.dataset.publicNav===page));
    window.scrollTo({top:0,behavior:'smooth'});
  };
  window.publicScroll=function(id){
    publicGo('home');
    requestAnimationFrame(()=>document.getElementById(id)?.scrollIntoView({behavior:'smooth',block:'start'}));
  };
  window.openAccessModal=function(){
    const modal=document.getElementById('access-modal');
    modal.classList.add('open');modal.setAttribute('aria-hidden','false');
    setTimeout(()=>document.getElementById('public-access-password')?.focus(),50);
  };
  window.closeAccessModal=function(){
    const modal=document.getElementById('access-modal');
    modal.classList.remove('open');modal.setAttribute('aria-hidden','true');
    const err=document.getElementById('access-error');if(err)err.style.display='none';
  };
  window.enterPlatform=function(e){
    e.preventDefault();
    const input=document.getElementById('public-access-password');
    if(input && input.value===ACCESS_PASSWORD){
      document.body.classList.add('app-unlocked');
      closeAccessModal();
      if(typeof shellNav==='function')shellNav('home');
    }else{
      const err=document.getElementById('access-error');if(err)err.style.display='block';
    }
  };
